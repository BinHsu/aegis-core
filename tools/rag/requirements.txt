# tools/rag/requirements.txt — ADR-0019 RAG seed pipeline.
#
# Pinned versions: reproducibility is load-bearing. If the embedding
# model version or the Qdrant client protocol changes under our feet,
# the "immutable corpus, reproducible index" promise quietly breaks.
# Bump these deliberately, not by `pip install -U`.
#
# Install into a venv inside the repo (CLAUDE.md Rule 6):
#     python3 -m venv .venv
#     source .venv/bin/activate
#     pip install -r tools/rag/requirements.txt

# --- Embeddings -------------------------------------------------------
# bge-m3 from BAAI, 2024 release. Dense + sparse + multi-vector from
# a single forward pass. Phase 3 uses dense only; later phases can
# enable hybrid retrieval without re-embedding.
FlagEmbedding==1.3.5

# Torch is FlagEmbedding's backbone; pin the exact major/minor so
# Apple Silicon MPS fallback behavior stays consistent across
# reviewer machines. CPU wheels keep the demo honest about local-only.
torch==2.6.0

# --- Vector store ----------------------------------------------------
# Qdrant Python client. Same API for local (file-backed) and cloud
# (QDRANT_URL + QDRANT_API_KEY). Pin to a release series that speaks
# the same gRPC protocol as the binary we ship against.
qdrant-client==1.12.1

# --- Chunking --------------------------------------------------------
# langchain-text-splitters supplies RecursiveCharacterTextSplitter
# with a Chinese-aware separator list. We depend on the split-out
# text-splitters package, not the full langchain umbrella, to keep
# the footprint small.
langchain-text-splitters==0.3.0

# --- Language detection ----------------------------------------------
# Used by the query-path (not by this seed script) to detect the
# tourist's query language and thread it into the LLM system prompt
# per ADR-0019 §Decision.6. Pinned here so the dev environment has
# one source of truth for every RAG-related dep.
langdetect==1.0.9
